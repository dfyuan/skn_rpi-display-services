436 char * skn_strip(char * alpha) {
437  if (alpha == NULL || strlen(alpha) < 1)
438  return alpha;
439 
440  int len = strlen(alpha);
441  int end = len - 1;
442  int start = 0;
443 
444  // use isgraph() or !isspace() vs isalnum() to allow ['|' ';' '%']
445  while (isgraph(alpha[end]) == 0 && end > 0) { // remove trailing non-alphanumeric chars
446  alpha[end--] = 0;
447  }
448 
449  len = strlen(alpha);
450  while ((isalnum(alpha[start]) == 0) && start < len) { // find first non-alpha stricter
451  start++;
452  }
453  if (start < len && start > 0) { // move in place
454  end = len - start;
455  memmove(&alpha[0], &alpha[start], end);
456  alpha[end] = 0;
457  }
458 
459  return alpha;
460 }

987 static int service_registry_response_parse(PServiceRegistry psreg, const char *response, int *errors) {
988  int control = 1;
989  char *base = NULL, *psep = NULL, *resp = NULL, *line = NULL,
990  *keypair = NULL, *element = NULL,
991  *name = NULL, *ip = NULL, *pport = NULL,
992  **meta = NULL;
993 
994  base = resp = strdup(response);
995 
996  if (strstr(response, "|")) {
997  psep = "|";
998  } else if (strstr(response, "%")) {
999  psep = "%";
1000  } else if (strstr(response, ";")) {
1001  psep = ";";
1002  }
1003 
1004  while ((line = strsep(&resp, psep)) != NULL) {
1005  if (strlen(line) < 16) {
1006  continue;
1007  }
1008 
1009  pport = ip = name = NULL;
1010  while ((keypair = strsep(&line, ",")) != NULL) {
1011  if (strlen(keypair) < 1) {
1012  continue;
1013  }
1014 
1015  control = 1;
1016  element = strstr(keypair, "=");
1017  if (element != NULL) {
1018  element[0] = 0;
1019  meta = service_registry_entry_create_helper(keypair, &name, &ip, &pport);
1020  if (meta != NULL && (element[1] != 0)) {
1021  *meta = skn_strip(++element);
1022  } else {
1023  control = 0;
1024  if (errors != NULL)
1025  (*errors)++;
1026  skn_logger(SD_WARNING, "Response format failure: for name=%s, ip=%s, port=%s, first failing entry: [%s]", name, ip, pport, keypair);
1027  break;
1028  }
1029  }
1030  } // end while line
1031 
1032  if (control == 1) { // catch a breakout caused by no value
1033  service_registry_entry_create(psreg, name, ip, pport, errors);
1034  }
1035 
1036  } // end while buffer
1037 
1038  free(base);
1039  return psreg->count;
1040 }
